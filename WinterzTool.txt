def clean_input(text):
    if not text:
        return ""
    text = str(text).strip()
    text = ''.join(c for c in text if c.isprintable() and ord(c) >= 32)
    return text.replace("^@", "").replace("^(", "").replace("�", "").replace("\x00", "")

def verify_token(token):
    headers = {"Authorization": token}
    try:
        r = requests.get("https://discord.com/api/v9/users/@me", headers=headers, timeout=10)
        if r.status_code == 200:
            user = r.json()
            print(f"{GREEN}✅ Valid Token → {user.get('username')}#{user.get('discriminator', '0000')}{RESET}")
            return True
        return False
    except:
        return False

def send_message(content, channel_id=None):
    if not channel_id: channel_id = CHANNEL_ID
    headers = {"Authorization": DISCORD_TOKEN, "Content-Type": "application/json"}
    try:
        r = requests.post(f"https://discord.com/api/v9/channels/{channel_id}/messages",
                         json={"content": content}, headers=headers, timeout=10)
        if r.status_code == 429:
            retry = r.json().get('retry_after', 1.5)
            print(f"{YELLOW}[!] Rate Limit - Waiting {retry}s{RESET}")
            time.sleep(retry)
            return False
        return r.status_code in [200, 204]
    except:
        return False

            except: pass
        requests.patch(f"https://discord.com/api/v9/guilds/{target_id}", json=data, headers=headers)
        print(f"{GREEN}✓ Name and icon updated{RESET}")

        print(f"{YELLOW}[2/5] Cloning roles with permissions...{RESET}")
        roles = requests.get(f"https://discord.com/api/v9/guilds/{source_id}/roles", headers=headers).json()
        role_map = {}
        for role in roles:
            if role['name'] == "@everyone":
                role_map[role['id']] = role['id']
                continue
            try:
                payload = {
                    "name": role['name'],
                    "color": role.get('color', 0),
                    "hoist": role.get('hoist', False),
                    "mentionable": role.get('mentionable', False),
                    "permissions": str(role.get('permissions', "0"))
                }
                new_role = requests.post(f"https://discord.com/api/v9/guilds/{target_id}/roles", json=payload, headers=headers).json()
                role_map[role['id']] = new_role['id']
            except: pass
            time.sleep(0.3)
        print(f"{GREEN}✓ Roles cloned{RESET}")

        # 3. Channels (Categories + Text + Voice)
        print(f"{YELLOW}[3/5] Cloning categories and channels...{RESET}")
        channels = requests.get(f"https://discord.com/api/v9/guilds/{source_id}/channels", headers=headers).json()
        channel_map = {}

        # Categories
        for ch in [c for c in channels if c['type'] == 4]:
            try:
                new_ch = requests.post(f"https://discord.com/api/v9/guilds/{target_id}/channels",
                                     json={"name": ch['name'], "type": 4, "position": ch.get('position', 0)}, headers=headers).json()
                channel_map[ch['id']] = new_ch['id']
            except: pass
            time.sleep(0.4)

        # Text and Voice
        for ch in [c for c in channels if c['type'] in [0, 2]]:
            try:
                payload = {
                    "name": ch['name'],
                    "type": ch['type'],
                    "position": ch.get('position', 0),
                    "parent_id": channel_map.get(ch.get('parent_id')),
                    "topic": ch.get('topic')
                }
                new_ch = requests.post(f"https://discord.com/api/v9/guilds/{target_id}/channels", json=payload, headers=headers).json()
                channel_map[ch['id']] = new_ch['id']
            except: pass
            time.sleep(0.35)
        print(f"{GREEN}✓ Channels and categories cloned{RESET}")

        # 4. Emojis
        print(f"{YELLOW}[4/5] Cloning emojis...{RESET}")
        emojis = requests.get(f"https://discord.com/api/v9/guilds/{source_id}/emojis", headers=headers).json()

                emoji_url = f"https://cdn.discordapp.com/emojis/{emoji['id']}.png"
                data = requests.get(emoji_url).content
                requests.post(f"https://discord.com/api/v9/guilds/{target_id}/emojis",
                            data={"name": emoji['name']}, files={'image': data}, headers={"Authorization": DISCORD_TOKEN})
            except: pass
            time.sleep(0.6)
        print(f"{GREEN}✓ Emojis cloned{RESET}")
